#### [234. 回文链表](https://leetcode-cn.com/problems/palindrome-linked-list/)

### 1.辅助栈

~~~c++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    bool isPalindrome(ListNode* head) {
        if(head == nullptr){
            return true;
        }
        stack<ListNode*> s;
        ListNode* curr = head;
        while(curr!=nullptr){
            s.push(curr);
            curr = curr->next;
        }
        curr = head;
        while(curr!=nullptr){
            if(curr->val != s.top()->val){
                return false;
            }
            s.pop();
            curr = curr->next;
        }
        return true;
    }
};
~~~

##### 时间：O(N)

##### 空间：O(N)



### 2.快慢指针

**基本思路：** 

利用快慢指针，找到中点位置，然后把中点右侧的链表进行反转（中间点指向空）；

比较前半段从左到右和后半段从右到左；

找到后要把原始链表反转回去。

~~~c++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    bool isPalindrome(ListNode* head) {
        if(head==nullptr || head->next==nullptr){
            return true;
        }
        //定义快慢指针
        ListNode* fast = head;
        ListNode* slow = head;

        //找中点
        while(fast->next!=nullptr && fast->next->next!=nullptr){
            fast = fast->next->next;
            slow = slow->next;
        }

        //反转单链表
        ListNode* pre = slow;
        ListNode* curr = slow->next;
        ListNode* nex = nullptr;
        slow->next = nullptr;

        while(curr!=nullptr){
            nex = curr->next;

            curr->next = pre;
            pre = curr;

            curr = nex;
        }

        //记录反转回去所需的节点
        ListNode* currRe = pre; // 头节点
        ListNode* end = slow; // 尾节点


        bool ans = true; // 因为后面需要反转回去，再返回结果，所以需要一个变量记录结果

        curr = head;
        while(curr!=nullptr && pre!=nullptr){
            if(curr->val != pre->val){
                ans = false;
                break;
            }
            curr = curr->next;
            pre = pre->next;
        }

        ListNode* preRe = nullptr;
        ListNode* nexRe = nullptr;
        // 反转回去
        while(currRe != end){
            nexRe = currRe->next;

            currRe->next = preRe;
            preRe = currRe;

            currRe = nexRe;
        }
        return ans;
    }
};
~~~

##### 时间：O(N)

##### 空间：O(1)
